function analyzeURL() { const url = document.getElementById('url-input').value.trim(); const explanationList = document.getElementById('explanation-list'); explanationList.innerHTML = ""; if (!url) return; let domain = ""; let tld = ""; let subdomains = 0; let length = 0; let score = 0; try { // Try normal parsing const parsed = new URL(url); domain = parsed.hostname; } catch (e) { // Fallback: extract domain manually addReason("✔ Invalid URL format (suspicious)"); let cleaned = url.replace("https://","").replace("http://",""); domain = cleaned.split("/")[0]; score += 25; // BIG penalty for invalid URL } // Continue analysis EVEN if parsing failed tld = domain.split('.').pop(); subdomains = domain.split('.').length - 2; length = domain.length; document.getElementById('domain-length').textContent = length; document.getElementById('subdomain-count').textContent = subdomains; document.getElementById('tld').textContent = "." + tld; // Now apply your rules if (length > 30) { score += 15; addReason("✔ URL too long"); } if (subdomains > 2) { score += 10; addReason("✔ Too many subdomains"); } if (["tk","ml","ga","cf","gq"].includes(tld)) { score += 20; addReason("✔ Suspicious TLD"); } if (url.includes("@")) { score += 15; addReason("✔ Contains @ symbol"); } if (url.includes("verify")) { score += 10; addReason("✔ Suspicious keyword"); } // Final cap if (score > 100) score = 100; updateRisk(score); function addReason(text) { const li = document.createElement("li"); li.textContent = text; explanationList.appendChild(li); } }
